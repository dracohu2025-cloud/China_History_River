# configuration file /etc/nginx/sites-enabled/history_river:
server {
    server_name history.aigc.green;

    # Frontend Static Files
    location / {
        root /home/ubuntu/history_river_2025/history_river_November_2025/history_river/dist;
        index index.html;
        try_files $uri $uri/ /index.html;
    }

    # Express Backend Proxy (API)
    location /api {
        proxy_pass http://127.0.0.1:4000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # Django Backend Proxy (Admin & Timeline API)
    location /admin {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location /timeline-api {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Django Static Files
    location /static/ {
        alias /home/ubuntu/history_river_2025/history_river_November_2025/history_river/dj_backend/staticfiles/;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }

    # Django Media Files
    location /media/ {
        alias /home/ubuntu/history_river_2025/history_river_November_2025/history_river/dj_backend/media/;
        expires 7d;
    }
}
